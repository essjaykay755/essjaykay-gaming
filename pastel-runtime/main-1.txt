let audioContext = null;
let soundEnabled = true;
function ensureAudio() {
  if (!soundEnabled) return null;
  if (!audioContext) audioContext = new (window.AudioContext || window.webkitAudioContext)();
  if (audioContext.state === 'suspended') audioContext.resume();
  return audioContext;
}
function tone(freq, duration = .08, type = 'sine', volume = .055, slide = 0) {
  const ctx = ensureAudio();
  if (!ctx) return;
  const now = ctx.currentTime;
  const osc = ctx.createOscillator();
  const gain = ctx.createGain();
  osc.type = type;
  osc.frequency.setValueAtTime(freq, now);
  if (slide) osc.frequency.exponentialRampToValueAtTime(Math.max(30, freq + slide), now + duration);
  gain.gain.setValueAtTime(volume, now);
  gain.gain.exponentialRampToValueAtTime(.0001, now + duration);
  osc.connect(gain).connect(ctx.destination);
  osc.start(now);
  osc.stop(now + duration);
}
function noiseBurst(duration = .15, volume = .05) {
  const ctx = ensureAudio();
  if (!ctx) return;
  const samples = Math.floor(ctx.sampleRate * duration);
  const buffer = ctx.createBuffer(1, samples, ctx.sampleRate);
  const data = buffer.getChannelData(0);
  for (let i = 0; i < samples; i += 1) data[i] = (Math.random() * 2 - 1) * (1 - i / samples);
  const source = ctx.createBufferSource();
  const gain = ctx.createGain();
  gain.gain.value = volume;
  source.buffer = buffer;
  source.connect(gain).connect(ctx.destination);
  source.start();
}

function makePlane(texture, width, height, transparent = true) {
  const material = new THREE.MeshBasicMaterial({ map: texture, transparent, depthWrite: false, toneMapped: false });
  const mesh = new THREE.Mesh(new THREE.PlaneGeometry(width, height), material);
  return mesh;
}

function makeCircleTexture(size = 128) {
  const c = document.createElement('canvas');
  c.width = c.height = size;
  const ctx = c.getContext('2d');
  const g = ctx.createRadialGradient(size / 2, size / 2, 0, size / 2, size / 2, size / 2);
  g.addColorStop(0, 'rgba(255,255,255,.7)');
  g.addColorStop(.55, 'rgba(255,255,255,.18)');
  g.addColorStop(1, 'rgba(255,255,255,0)');
  ctx.fillStyle = g;
  ctx.fillRect(0, 0, size, size);
  const texture = new THREE.CanvasTexture(c);
  texture.colorSpace = THREE.SRGBColorSpace;
  return texture;
}

function makeSparkTexture(size = 64) {
  const c = document.createElement('canvas');
  c.width = c.height = size;
  const ctx = c.getContext('2d');
  ctx.translate(size / 2, size / 2);
  ctx.fillStyle = '#fff';
  ctx.beginPath();
  for (let i = 0; i < 8; i += 1) {
    const a = i * Math.PI / 4;
    const r = i % 2 ? size * .1 : size * .48;
    const x = Math.cos(a) * r;
    const y = Math.sin(a) * r;
    if (i === 0) ctx.moveTo(x, y); else ctx.lineTo(x, y);
  }
  ctx.closePath();
  ctx.fill();
  const texture = new THREE.CanvasTexture(c);
  texture.colorSpace = THREE.SRGBColorSpace;
  return texture;
}

function makeRing(color, inner = 44, outer = 50) {
  const material = new THREE.MeshBasicMaterial({ color, transparent: true, opacity: 1, depthWrite: false, blending: THREE.NormalBlending });
  const mesh = new THREE.Mesh(new THREE.RingGeometry(inner, outer, 48), material);
  mesh.visible = false;
  mesh.renderOrder = 5;
  world.add(mesh);
  rings.push({ mesh, life: 0, maxLife: .5, grow: 1, driftX: 0, driftY: 0 });
  return rings[rings.length - 1];
}
function spawnRing(x, y, color, options = {}) {
  let ring = rings.find((r) => r.life <= 0);
  if (!ring) ring = makeRing(color);
  ring.mesh.material.color.set(color);
  ring.mesh.material.opacity = options.opacity ?? .8;
  ring.mesh.scale.setScalar(options.scale ?? 1);
  setWorldPosition(ring.mesh, x, y, options.z ?? 5);
  ring.life = options.life ?? .42;
  ring.maxLife = ring.life;
  ring.grow = options.grow ?? 1.9;
  ring.driftX = options.driftX ?? 0;
  ring.driftY = options.driftY ?? 0;
  ring.mesh.visible = true;
}

const glowTexture = makeCircleTexture();
const sparkTexture = makeSparkTexture();
function spawnParticle(x, y, color, count = 18, force = 350, shape = 'spark') {
  for (let i = 0; i < count; i += 1) {
    const angle = Math.random() * Math.PI * 2;
    const speed = force * (.25 + Math.random() * .85);
    const size = 9 + Math.random() * 20;
    const mat = new THREE.SpriteMaterial({
      map: shape === 'glow' ? glowTexture : sparkTexture,
      color,
      transparent: true,
      opacity: .95,
      depthWrite: false,
      blending: THREE.NormalBlending,
    });
    const sprite = new THREE.Sprite(mat);
    sprite.scale.set(size, size, 1);
    setWorldPosition(sprite, x, y, 8);
    world.add(sprite);
    particles.push({ sprite, x, y, vx: Math.cos(angle) * speed, vy: Math.sin(angle) * speed, gravity: 180 + Math.random() * 180, life: .42 + Math.random() * .72, maxLife: 1, spin: (Math.random() - .5) * 11 });
  }
}

function createActor(texture, color, side) {
  const group = new THREE.Group();
  const visualW = texture.image?.width || 96;
  const visualH = texture.image?.height || 96;
  group.renderOrder = 7;
  const shadowMat = new THREE.SpriteMaterial({ map: shadowTexture, color: 0x316c48, transparent: true, opacity: .28, depthWrite: false });
  const shadow = new THREE.Sprite(shadowMat);
  shadow.scale.set(visualW * .9, visualH * .56, 1);
  shadow.position.set(8, -13, -1);
  group.add(shadow);

  const mesh = makePlane(texture, visualW, visualH, true);
  mesh.renderOrder = 8;
  mesh.material.color.set(0xffffff);
  group.add(mesh);

  const orbit = new THREE.Group();
  orbit.visible = false;
  for (let i = 0; i < 3; i += 1) {
    const curve = new THREE.EllipseCurve(0, 0, 61 + i * 5, 39 + i * 4, 0, Math.PI * (1.45 + i * .13), false, i * .55);
    const points = curve.getPoints(64).map((p) => new THREE.Vector3(p.x, p.y, 0));
    const geo = new THREE.BufferGeometry().setFromPoints(points);
    const mat = new THREE.LineBasicMaterial({ color: 0xffffff, transparent: true, opacity: .78 - i * .15, depthWrite: false });
    const line = new THREE.Line(geo, mat);
    line.rotation.z = i * 1.8;
    orbit.add(line);
  }
  orbit.position.z = 2;
  group.add(orbit);
  world.add(group);
  return { group, mesh, shadow, orbit, color: new THREE.Color(color), side, squashX: 1, squashY: 1, orbitSpin: 0, visualW, visualH };
}

function createPuck(texture) {
  const group = new THREE.Group();
  const visualW = texture.image?.width || 118;
  const visualH = texture.image?.height || 118;
  const shadowMat = new THREE.SpriteMaterial({ map: shadowTexture, color: 0x286944, transparent: true, opacity: .25, depthWrite: false });
  const shadow = new THREE.Sprite(shadowMat);
  shadow.scale.set(visualW * .8, visualH * .46, 1);
  shadow.position.set(7, -11, -1);
  group.add(shadow);
  const mesh = makePlane(texture, visualW, visualH, true);
  mesh.material.color.copy(puck.tint);
  mesh.renderOrder = 9;
  group.add(mesh);
  world.add(group);
  return { group, mesh, shadow, squashX: 1, squashY: 1, visualW, visualH };
}

function spawnAfterimage(actor, color) {
  if (!actor?.mesh?.material?.map) return;
  const mat = new THREE.MeshBasicMaterial({ map: actor.mesh.material.map, color, transparent: true, opacity: .38, depthWrite: false, toneMapped: false });
  const mesh = new THREE.Mesh(actor.mesh.geometry, mat);
  mesh.scale.copy(actor.mesh.scale);
  mesh.rotation.copy(actor.group.rotation);
  setWorldPosition(mesh, actor.group.position.x, WORLD_H - actor.group.position.y, 4);
  world.add(mesh);
  afterimages.push({ mesh, life: .32, maxLife: .32 });
}
